# YTableComponents
these are just regular table components with some default class names that set up some default stylings.

this has an example sortability on the first column.

<Playground>
<State initialState={{ sort: undefined, sortClass: ''}}>
<Observe>
{({ sort, sortClass, setState }) => (
        <YTable>
        <YThead>
            <YTr>
                <YTh resizable={true}>
                    Col A
                    <YThSortIndicator
                        onClick={() => setState({
                            sort: (sortClass === 'desc'
                                ? (a, b) => b.key - a.key
                                : (a, b) => a.key - b.key),
                            sortClass: (sortClass === 'desc'
                                ? 'asc'
                                : 'desc')
                        })}
                        className={sortClass}
                    />
                </YTh>
                <YTh>
                    Col B
                </YTh>
                <YTh>
                    Col C
                </YTh>
                <YTh>
                    Col D
                </YTh>
            </YTr>
        </YThead>
        <YTBody>
            {[
                {key: 3, prefix: 'B'},
                {key: 73, prefix: 'Z'},
                {key: 1, prefix: 'B'},
                {key: 22, prefix: 'Z'}
            ].sort(sort).map(
                elem => <YTr key={elem.key}>
                    {['EEP', 'OOP', 'EEP', 'OOP'].map((suffix, i) => <YTd key={suffix + i}>{i === 0 ? elem.key : elem.prefix + suffix}</YTd>)}
                </YTr>
            )}
        </YTBody>
    </YTable>
) }
</Observe>
</State>
</Playground>
