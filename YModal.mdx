## position: fixed (default)
<Playground>
<State initialState={{ count: false }}>
<Observe>
{({ count, setState }) => (
	<>
		<button onClick={() => setState(s => ({ count: true }))}>
		Click me
		</button>
		{count && <YModal>Hello, I am a basic modal, close me with <YButton onClick={() => setState(s => ({count: false}))}>this button</YButton></YModal>}
	</>
) }
</Observe>
</State>
</Playground>


## position: fixed (default) with height overflow
<Playground>
<State initialState={{ count: false }}>
<Observe>
{({ count, setState }) => (
	<>
		<button onClick={() => setState(s => ({ count: true }))}>
		Click me
		</button>
		{count && <YModal>
			Hello, I am a basic modal,
			<div>
				{Array(100).fill().map(i => <span key={'key'+i}>and I'm just way too verbose</span>)}
			</div>
			{Array(100).fill().map(i => <div key={'key'+i}>and I'm just way too verbose</div>)}
			, close me with
			
			<YButton onClick={() => setState(s => ({count: false}))}>this button</YButton></YModal>}
	</>
) }
</Observe>
</State>
</Playground>
