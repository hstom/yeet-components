### position: fixed (default)
<Playground>
<State initialState={{ count: false }}>
<Observe>
{({ count, setState }) => (
	<>
		<YButtonPrimary onClick={() => setState(s => ({ count: true }))}>
		Click me
		</YButtonPrimary>
		{count && <YModal>Hello, I am a basic modal, close me with <YButtonPrimary onClick={() => setState(s => ({count: false}))}>this button</YButtonPrimary></YModal>}
	</>
) }
</Observe>
</State>

## position: fixed (default) with height overflow
<Playground>

	<>
		<YButtonPrimary onClick={() => setState(s => ({ count: true }))}>
		Click me
		</YButtonPrimary>
		{count && <YModal>
			Hello, I am a basic modal,
			<div>
				{Array(100).fill().map((_, i) => <span key={'key'+i}>and I'm just way too verbose</span>)}
			</div>
			{Array(100).fill().map((_, i) => <div key={'key'+i}>and I'm just way too verbose</div>)}

			<YButtonPrimary onClick={() => setState(s => ({count: false}))}>this button</YButtonPrimary></YModal>}
	</>
) }
</Observe>
</State>
</Playground>
