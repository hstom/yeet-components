{() => {
	const [open, setOpen] = useState(false);
	return (
	<>
		<pre style={{margin: 0}}>{'form with default submit, cancel, and outside click'}</pre>
		<YButtonPrimary onClick={() => setOpen(!open)}>
			Click me
		</YButtonPrimary>
		{
			open &&
			<YModalForm
				title='Modal Title'
				submitAction={() => setOpen(false)}
				cancelAction={() => setOpen(false)}
			>
				Hello, I am a basic modal, close me with the standard button!
			</YModalForm>
		}
	</>
	);
}}
