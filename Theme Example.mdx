# Themed YModalForm

<Playground>
<State initialState={{ count: false }}>
<Observe>
{({ count, setState }) => (
	<>
		<button onClick={() => setState(s => ({ count: true }))}>
		Click me
		</button>
		{count &&
			<YModalForm
				title='Modal Title'
				submitAction={() => setState(s => ({count: false}))}
				cancelAction={() => setState(s => ({count: false}))}
			>
				Hello, I am a basic modal, close me with the standard button!
			</YModalForm>
		}
	</>
) }
</Observe>
</State>
</Playground>
